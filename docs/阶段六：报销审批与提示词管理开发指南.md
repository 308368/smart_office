**OfExpenseServiceImpl.java** - office-service/.../service/impl/
```java
@Service
@RequiredArgsConstructor
public class OfExpenseServiceImpl extends ServiceImpl<OfExpenseMapper, OfExpense>
    implements IOfExpenseService {

    private final OfExpenseMapper expenseMapper;
    private final OfExpenseItemMapper itemMapper;

    @Override
    public ExpenseVO addExpense(AddExpenseDTO dto, SysUser user, DeptVo dept) {
        // 1. 计算总额
        BigDecimal totalAmount = dto.getItems().stream()
            .map(ExpenseItemDTO::getAmount)
            .reduce(BigDecimal.ZERO, BigDecimal::add);

        // 2. 生成报销编号 EX + 时间戳
        String expenseNo = "EX" + System.currentTimeMillis();

        // 3. 保存报销主表
        OfExpense expense = new OfExpense();
        expense.setExpenseNo(expenseNo);
        expense.setUserId(user.getId());
        expense.setUserName(user.getUsername());
        expense.setDeptId(dept.getId());
        expense.setDeptName(dept.getName());
        expense.setExpenseType(dto.getExpenseType());
        expense.setTotalAmount(totalAmount);
        expense.setDescription(dto.getDescription());
        expense.setStatus(ExpenseEnum.PENDING.getCode());
        expenseMapper.insert(expense);

        // 4. 保存费用明细
        for (ExpenseItemDTO itemDTO : dto.getItems()) {
            OfExpenseItem item = new OfExpenseItem();
            item.setExpenseId(expense.getId());
            item.setItemType(itemDTO.getItemType());
            item.setAmount(itemDTO.getAmount());
            item.setDescription(itemDTO.getDescription());
            item.setExpenseDate(itemDTO.getExpenseDate());
            item.setReceiptUrl(itemDTO.getReceiptUrl());
            itemMapper.insert(item);
        }

        return getExpenseDetail(expense.getId());
    }

    @Override
    public PageResult<ExpenseVO> getMyExpenseList(ExpenseQueryParam param, String username) {
        LambdaQueryWrapper<OfExpense> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(OfExpense::getUserName, username)
               .orderByDesc(OfExpense::getCreateTime);
        // 分页查询...

        // 转换并返回
        return PageResult.of(records, total, ExpenseVO.class);
    }

    @Override
    public ExpenseVO getExpenseDetail(Long id) {
        OfExpense expense = expenseMapper.selectById(id);
        if (expense == null) {
            throw new RuntimeException("报销记录不存在");
        }

        ExpenseVO vo = new ExpenseVO();
        // 属性拷贝...

        // 查询费用明细
        List<OfExpenseItem> items = itemMapper.selectList(
            new LambdaQueryWrapper<OfExpenseItem>().eq(OfExpenseItem::getExpenseId, id)
        );
        vo.setItems(items.stream().map(this::toItemVO).toList());

        return vo;
    }

    @Override
    public PageResult<ExpenseVO> getPendingList(ExpenseQueryParam param) {
        // 查询 status = 1（待审批）的记录
        // 分页、排序、返回
    }

    @Override
    public void approveExpense(Long id, ApproveDTO dto, SysUser approver) {
        OfExpense expense = expenseMapper.selectById(id);
        if (expense == null || expense.getStatus() != ExpenseEnum.PENDING.getCode()) {
            throw new RuntimeException("报销记录不存在或状态不允许审批");
        }

        expense.setApproverId(approver.getId());
        expense.setApproverName(approver.getUsername());
        expense.setApproveTime(LocalDateTime.now());
        expense.setApproveComment(dto.getComment());
        expense.setStatus(dto.getApproved() ?
            ExpenseEnum.APPROVED.getCode() : ExpenseEnum.REJECTED.getCode());

        expenseMapper.updateById(expense);
    }

    @Override
    public void cancelExpense(Long id, String username) {
        OfExpense expense = expenseMapper.selectById(id);
        if (expense == null) {
            throw new RuntimeException("报销记录不存在");
        }
        if (!expense.getUserName().equals(username)) {
            throw new RuntimeException("只能取消自己的报销申请");
        }
        if (expense.getStatus() != ExpenseEnum.PENDING.getCode()) {
            throw new RuntimeException("只能取消待审批状态的报销");
        }

        expense.setStatus(ExpenseEnum.CANCELLED.getCode());
        expenseMapper.updateById(expense);
    }
}
```

**OfExpenseController.java** - office-service/.../controller/

```java
@RequestMapping("/office/expense")
@RequiredArgsConstructor
public class OfExpenseController {

    private final IOfExpenseService expenseService;

    /**
     * 创建报销申请
     */
    @PostMapping("/create")
    @PreAuthorize("hasAuthority('expense:add')")
    public Result<ExpenseVO> create(@RequestBody @Valid AddExpenseDTO dto) {
        SysUser user = getCurrentUser();
        DeptVo dept = getCurrentDept();
        return Result.success(expenseService.addExpense(dto, user, dept));
    }

    /**
     * 我的报销列表
     */
    @GetMapping("/list")
    public Result<PageResult<ExpenseVO>> list(ExpenseQueryParam param) {
        SysUser user = getCurrentUser();
        return Result.success(expenseService.getMyExpenseList(param, user.getUsername()));
    }

    /**
     * 报销详情
     */
    @GetMapping("/{id}")
    public Result<ExpenseVO> detail(@PathVariable Long id) {
        return Result.success(expenseService.getExpenseDetail(id));
    }

    /**
     * 待审批列表
     */
    @GetMapping("/pending")
    @PreAuthorize("hasAuthority('expense:approve')")
    public Result<PageResult<ExpenseVO>> pending(ExpenseQueryParam param) {
        return Result.success(expenseService.getPendingList(param));
    }

    /**
     * 审批报销
     */
    @PutMapping("/approve/{id}")
    @PreAuthorize("hasAuthority('expense:approve')")
    public Result approve(@PathVariable Long id, @RequestBody @Valid ApproveDTO dto) {
        SysUser approver = getCurrentUser();
        expenseService.approveExpense(id, dto, approver);
        return Result.success();
    }

    /**
     * 取消报销
     */
    @PutMapping("/cancel/{id}")
    public Result cancel(@PathVariable Long id) {
        SysUser user = getCurrentUser();
        expenseService.cancelExpense(id, user.getUsername());
        return Result.success();
    }
}
```

```java
@Service
@RequiredArgsConstructor
public class ChatPromptServiceImpl extends ServiceImpl<ChatPromptMapper, ChatPrompt>
    implements IChatPromptService {

    @Override
    public List<PromptVO> getAvailablePrompts(Long userId) {
        LambdaQueryWrapper<ChatPrompt> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(ChatPrompt::getIsPublic, 1)  // 公共的
                .or()
                .eq(ChatPrompt::getCreateBy, userId)  // 或自己创建的
                .orderByDesc(ChatPrompt::getCreateTime);

        return list(wrapper).stream().map(this::toVO).toList();
    }

    @Override
    public PromptVO addPrompt(AddPromptDTO dto, Long userId) {
        ChatPrompt prompt = new ChatPrompt();
        prompt.setName(dto.getName());
        prompt.setDescription(dto.getDescription());
        prompt.setPrompt(dto.getPrompt());
        prompt.setCategory(dto.getCategory());
        prompt.setIsPublic(dto.getIsPublic());
        prompt.setCreateBy(userId);
        baseMapper.insert(prompt);
        return toVO(prompt);
    }

    @Override
    public void updatePrompt(Long id, AddPromptDTO dto) {
        ChatPrompt prompt = baseMapper.selectById(id);
        if (prompt == null) throw new RuntimeException("提示词不存在");

        prompt.setName(dto.getName());
        prompt.setDescription(dto.getDescription());
        prompt.setPrompt(dto.getPrompt());
        prompt.setCategory(dto.getCategory());
        prompt.setIsPublic(dto.getIsPublic());
        baseMapper.updateById(prompt);
    }

    @Override
    public void deletePrompt(Long id) {
        baseMapper.deleteById(id);
    }
}
```

```java
@Component
@RequiredArgsConstructor
public class WebSocketService {
    private final SimpMessagingTemplate simpMessagingTemplate;

    /**
     * 发送消息到指定主题
     */
    public void sendToTopic(String destination, Map<String, Object> message) {
        simpMessagingTemplate.convertAndSend(destination, message);
    }

    /**
     * 文档分块完成通知
     */
    public void sendChunkComplete(Long kbId, Long docId, String docTitle, int chunkCount) {
        Map<String, Object> msg = new HashMap<>();
        msg.put("type", "chunk_complete");
        msg.put("kbId", kbId);
        msg.put("docId", docId);
        msg.put("docTitle", docTitle);
        msg.put("chunkCount", chunkCount);
        msg.put("timestamp", System.currentTimeMillis());
        simpMessagingTemplate.convertAndSend("/topic/document", msg);
    }
}
```
